'use strict';
 
var $ = require('jquery');
var autocomplete = require('jquery-ui/autocomplete');
var utils = require('./utils');
 
function Autocomplete(elems) {
  var _this = this;
 
  var self = this;
 
  function getInputVal() {
    return self.$input.val();
  };
 
  function getRelevantListItems(request, response) {
    var serviceUrl = getServiceUrl(request.term);
    var formattedData = void 0;
 
    $.ajax({
      url: serviceUrl,
      dataType: 'jsonp',
      type: 'GET',
      success: function success(data) {
        formattedData = formatAjaxResponse(data, request.term);
        response(formattedData);
      },
      error: function error(err) {
        console.error(err);
      }
    });
  };
 
  function formatAjaxResponse(data, term) {
 
    var results = [];
 
    $.each(data, function (i, val) {
      val.map(function (obj) {
        return obj.queryTerm = term;
      });
 
      results = results.concat(val);
    });
 
    return results;
  };
 
  function getServiceUrl(queryTerm) {
 
    var isOmnibox = self.type === 'omnibox' ? true : false,
        includeShadyGigs = self.includeShadyGigs ? true : false,
        protocol = "https://fiverr.com",
        omniboxUrlPrefix = '/search/omnibox',
        searchUrlPrefix = '/search/autocomplete',
        autoCompleteUrlPrefix = isOmnibox ? omniboxUrlPrefix : searchUrlPrefix,
        shadyGigsParam = includeShadyGigs ? '&shady_gigs=true' : '',
        serviceUrl = '' + protocol + autoCompleteUrlPrefix + '?locale=' + self.locale + '&query=' + queryTerm + shadyGigsParam;
 
    return serviceUrl;
  };
 
  function handleAutocompleteItemSelect(e) {
    if (e) e.preventDefault();
 
    self.$form.submit();
  };
 
  function handleInputFocus(e) {
    if (e) e.preventDefault();
 
    var inputVal = getInputVal();
 
    if (inputVal.length > 1) {
      self.$dropdown.show();
    }
  };
 
  function getAutocompleteOptions() {
    var options = {};
 
    options.source = getRelevantListItems;
    options.minLength = 2;
    options.appendTo = self.$form;
    options.messages = {
      noResults: '',
      results: function results() {}
    };
 
    return options;
  };
 
  function attachAutocompletePlugin() {
    var options = getAutocompleteOptions();
 
    self.$input.autocomplete(options).data('ui-autocomplete')._renderItem = utils.pluginUtils.handleListItemCreation;
 
    self.$dropdown = self.$form.find('.ui-autocomplete');
  };
 
  function bindEventListeners() {
    self.$input.on('focus', handleInputFocus);
 
    self.$dropdown.on('click', handleAutocompleteItemSelect);
  };
 
  return {
 
    init: function init() {
 
      _this.$form = elems.form;
      _this.$input = elems.input;
      _this.type = _this.$input.data('component-type');
      _this.locale = _this.$input.data('component-locale');
      _this.includeShadyGigs = _this.$input.data('component-include-shady-gigs');
 
      attachAutocompletePlugin();
      bindEventListeners();
    }
 
  };
};
 
module.exports = Autocomplete;
